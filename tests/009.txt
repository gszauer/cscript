void RunVariableTests() {
	int i = 7;
	float f = 2.6;
	bool t = true;
	bool _f = false;
	char z = '0';
	object n = null;
	Vec3 v = Vec3();

	PrintInt(n as int, 0);
	PrintInt(v as int, 0);

	PrintFloat(i as float, 7.0);
	PrintFloat(f as float, 2.6);
	PrintFloat(t as float, 1.0);
	PrintFloat(_f as float, 0.0);
	PrintFloat(z as float, 48.0);
	PrintFloat(n as float, 0.0);
	PrintFloat(v as float, 0.0);
	print('\n');

	int n7 = 97;
	int n8 = 98;
	int n9 = 99;
	char a = 'a';

	// Literals to char
	PrintChar(n7 as char, 'a');
	PrintChar(n8 as char, 'b');
	PrintChar((true as int + n8) as char, 'c');
	PrintChar((false as int + n8) as char, 'b');
	PrintChar(a as char, 'a');
	PrintChar(t as char, 't');
	PrintChar(_f as char, '\0'); // false to char
	PrintChar(n as char, '\0');
	PrintChar(v as char, '\0');
	print('\n');

	int _z = 0;
	float __f = 0.0;
	char _t = 't';
	char ___f = 'f';

	// Literals to bool
	PrintBool(i as bool, true);
	PrintBool(_z as bool, false);
	PrintBool(f as bool, true);
	PrintBool(__f as bool, false);
	PrintBool(t as bool, true);
	PrintBool(_f as bool, false);
	PrintBool('\0' as bool, false);
	PrintBool(z as bool, true);
	PrintBool(_t as bool, true);
	PrintBool(___f as bool, true);
	PrintBool(n as bool, false);
	PrintBool(v as bool, false);
	print('\n');
}

struct Vec3 {
	float x;
	float y;
	float z;
}

struct Vec2 {
	float x;
	float y;
}

void PrintObject(object o) {
	print(o);print('\n');
	print(type(o));print('\n');
	print(type( dynamic o));print('\n');
}

void PrintNull(object o) {
	print (o);print(' ');print('/');print(' ');
	print(null);print('\n');
}

void PrintBool(bool b, bool e) {
	print(b);print(' ');print('/');print(' ');print(e);
	print('\n');
}

void PrintInt(int a, int b) {
	print(a);print(' ');print('/');print(' ');print(b);
	print('\n');
}

void PrintFloat(float a, float b) {
	print(a);print(' ');print('/');print(' ');print(b);
	print('\n');
}

void PrintChar(char a, char b) {
	print(a);print(' ');print('/');print(' ');print(b);
	print('\n');
}

Vec3 newVec3(float x, float y, float z) {
	Vec3 v = Vec3();
	v.x = x;
	v.y = y;
	v.z = z;
	return v;
}

}
